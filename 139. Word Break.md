Problem :

```
Given a string s and a dictionary of strings wordDict, return true if s can be segmented into a space-separated sequence of one or more dictionary words.

Note that the same word in the dictionary may be reused multiple times in the segmentation.

 

Example 1:

Input: s = "leetcode", wordDict = ["leet","code"]
Output: true
Explanation: Return true because "leetcode" can be segmented as "leet code".
Example 2:

Input: s = "applepenapple", wordDict = ["apple","pen"]
Output: true
Explanation: Return true because "applepenapple" can be segmented as "apple pen apple".
Note that you are allowed to reuse a dictionary word.
Example 3:

Input: s = "catsandog", wordDict = ["cats","dog","sand","and","cat"]
Output: false
 

Constraints:

1 <= s.length <= 300
1 <= wordDict.length <= 1000
1 <= wordDict[i].length <= 20
s and wordDict[i] consist of only lowercase English letters.
All the strings of wordDict are unique.
 
```

class Trie{
    Trie[] children;
    boolean isWord;
    Trie(){
        this.children = new Trie[26];
    }
    void addWord(String word){
        Trie iter = this;
        for(char ch : word.toCharArray()){
            int idx = ch-'a';
            if(iter.children[idx]==null)
                iter.children[idx] = new Trie();
            iter = iter.children[idx];
        }
        iter.isWord = true;
    }
}
class Solution {
    public boolean wordBreak(String s, List<String> wordDict) {
        int len = s.length();
        Trie trie = new Trie();
        for(String word : wordDict){
            trie.addWord(word);
        }
        boolean[] mat = new boolean[len+1];
        mat[0] = true;
        for(int i=0;i<=len;i++){
            if(!mat[i]) continue;
            Trie iter = trie;
            for(int j=i;j<len;j++){
                int idx = s.charAt(j)-'a';
                if(iter.children[idx]==null) break;
                iter = iter.children[idx];
                if(iter.isWord) mat[j+1] = true;
            }
        }
        return mat[len];
    }
}

class Solution {
    public boolean wordBreak(String s, List<String> wordDict) {
        Set<String> words = new HashSet<>(wordDict);
        int len = s.length();
        int max = 0,min = 300;
        for(String word : wordDict){
            max = Math.max(max,word.length());
            min = Math.min(min,word.length());
        }
        boolean[] dp = new boolean[len+1];
        dp[0] = true;
        for(int i=0;i<len+1;i++){
            if(!dp[i]) continue;
            int cmax = Math.min(i+max,len);
            for(int j=i+min;j<=cmax;j++){
                if(words.contains(s.substring(i,j))){
                    dp[j] = true;
                }
            }
        }
        return dp[len];
    }
}
